function wrap(type)
   local sides = redstone.getSides()
   for i=1,6 do
      if peripheral.getType(sides[i]) == type then
         return peripheral.wrap(sides[i])
      end
   end
   return nil
end

-- Exemples :
m = wrap("monitor")
c = wrap("computer")
